#include <stdio.h>

int fun(int x)
{
    int p;
    if (x == 0 || x == 1)
        return (3);
    p = x - fun(x - 2);
    return p;
}
int main()
{
    printf("%d\n", fun(7));
}

#include <stdio.h>
void get_put()
{
    char ch;
    ch = getchar(); 
    if (ch != '\n')
        get_put();  //递归逆序
    putchar(ch);
}
int main()
{
    get_put();
    printf("\n");
}

// 1234  ->  4321
